\begin{frame}{Les formules}
	\underline{Loi d'Ampère} \\
	$$\oint \vec{H}\cdot\vec{dl}\cong H_m \cdot L + H_e \cdot e = N \cdot I$$ \\
	\center{Quand nous avons} $\ H_e \cong H_i$
	$$\oint \vec{H} \cdot \vec{dl} \cong H \cdot l = N \cdot I$$
	Dans le noyau $B_i=\mu_0 \cdot \mu_r \cdot H_i$ \\
	A la surface latérale $B_e = \mu_0 \cdot H_e$ \\
\end{frame}

\begin{frame}{Première situation : sans entrefer}
	\center{Hypothèse : La champ H et le champ B sont identiques partout. Dans un électroaimant sans entrefer 
					tout le champ magnétique B se retrouve dans le matériau.}
	\includegraphics[scale=0.5]{Circuitalpha.png}
	\center{\underline{Nos solutions :}}
	\center{Nous obtenons 188 spires et $H = 497,36 \frac{A}{m}$ pour :} \\
	$B=1T \ I=0.5A \ \mu_r=1600$
\end{frame}

\begin{frame}{Deuxième situation : entrefer de 1mm}
	\center{Hypothèse : Le champ H est uniforme. Dans un électroaimant avec entrefer tout le champ magnétisant H 
					se retrouve dans l'entrefer ($H_m = 0$).}\\
	\includegraphics[scale=0.5]{Circuitbeta}
	\center{\underline{Nos solutions :}}
	\center{Nous obtenons $H_e = 94000 \frac{H}{m}$ et $B_e=0.06T
	= B_m$ (par la conservation des flux) pour :} \\
	 $N = 188$ spires \ $e=1mm$ \ $I=0.5A$ \ $\mu_r = 1600$ 
\end{frame}

\begin{frame}{Troisième situation : entrefer de 11mm}
